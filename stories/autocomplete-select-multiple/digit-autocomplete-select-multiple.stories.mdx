# DigitAutocompleteSelectMultiple

Input to select multiple values via autocomplete when searching for options. 

<Preview>
    <Story name="Autocomplete multiple">
        {() => {
            const options = [
                { text: "The Dark Knight Rises", value: "the_dark"},
                { text: "Princess Mononoke", value: "princess" },
                { text: "Aliens", value: "aliens" },
                { text: "Oldboy" , value: "oldboy"},
                { text: "Once Upon a Time in America", value: "once" },
                { text: "Witness for the Prosecution", value: "witness" },
                { text: "Das Boot", value: "das", disabled: true},
            ]
            const [value, setValue] = useState(["aliens"]);
            return (
                <DigitAutocompleteSelectMultiple outlinedupperLabel="what" options={options} value={value} onChange={(e) => {setValue(e.target.value);}}/>        
            );
        }}
    </Story>
</Preview>
