# DigitAutocompleteSelectSingle

Input to select a single value via autocomplete when searching for the option. 

<Preview>
    <Story name="Autocomplete single">
        {() => {
            const options = [
                { text: "The Dark Knight Rises", value: "the_dark"},
                { text: "Princess Mononoke", value: "princess" },
                { text: "Aliens", value: "aliens" },
                { text: "Oldboy" , value: "oldboy"},
                { text: "Once Upon a Time in America", value: "once" },
                { text: "Witness for the Prosecution", value: "Witness" },
                { text: "Das Boot", value: "das", disabled: true},
            ]
            const [value, setValue] = useState("aliens");
            return (
                <DigitAutocompleteSelectSingle filled error errorMessage="hej" disabled upperLabel="what" options={options} value={value} onChange={(e) => {setValue(e.target.value);}}/>        
            );
        }}
    </Story>
