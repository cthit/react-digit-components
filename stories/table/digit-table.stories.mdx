## Large table

<Canvas>
    <Story name="large table">
        {() => {
            const data = useMemo(() => {
                const firstNames = ["Adam", "Anders", "Alf", "Amanda", "Anna"];
                const lastNames = [
                    "Adamsson",
                    "Andersson",
                    "Alfsson",
                    "Amandadotter",
                    "Annadotter"
                ];
                const output = [];
                for (let i = 0; i < 500; i++) {
                    output.push({
                        id: Math.random() + "",
                        firstName: firstNames[Math.floor(Math.random() * 4)],
                        lastName: lastNames[Math.floor(Math.random() * 4)],
                        age: Math.random() * 90
                    });
                }
                return output;
            });
            return (
                <DigitTable
                    titleText="Humans"
                    idProp="id"
                    startOrderBy="id"
                    columnsOrder={["id", "firstName", "lastName", "age"]}
                    headerTexts={{
                        id: "Id",
                        firstName: "First name",
                        lastName: "Last name",
                        age: "Age"
                    }}
                    data={data}
                    search
                />
            );
        }}
    </Story>
</Canvas>
