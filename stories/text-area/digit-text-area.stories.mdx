## How to use with DigitEditData

<Preview>
    <Story name="Text field with EditData">
        <DigitEditData
            hasButtons
            initialValues={{
                firstname: "",
                surname: "",
                other: ""
            }}
            onSubmit={(values, actions) => {
                console.log(values);
                actions.resetForm();
            }}
            validationSchema={yup.object().shape({
                firstname: yup.string().required(),
                surname: yup.string().required(),
                other: yup.string().required()
            })}
            titleText={"Title"}
            submitText={"Submit"}
            keysOrder={["firstname", "surname", "other"]}
            keysComponentData={{
                firstname: {
                    component: DigitTextArea,
                    componentProps: {
                        upperLabel: "Firstname"
                    }
                },
                surname: {
                    component: DigitTextArea,
                    componentProps: {
                        upperLabel: "Surname"
                    }
                },
                other: {
                    component: DigitTextArea,
                    componentProps: {
                        upperLabel: "Other",
                        outlined: true
                    }
                }
            }}
            submitButton={{    
                text: "Skicka",
            }}
        />
    </Story>
</Preview>
